Zapytanie 3:
SELECT p.rok, AVG(s.dlugosc) AS `Średnia długość` FROM parada p JOIN udzial u ON p.id = u.id_parada JOIN smok s ON u.id_smok = s.id WHERE CAST(p.rok AS UNSIGNED) > 2005 GROUP BY p.rok ORDER BY p.rok;
