Zapytanie 2: 
SELECT producent, model, sezon, cena FROM opony WHERE nr_kat = 9;

Zapytanie 3: 
SELECT z.id_zam, z.ilosc, o.model, o.cena 
FROM zamowienie z 
JOIN opony o ON z.nr_kat = o.nr_kat 
ORDER BY RAND() 
LIMIT 1;
