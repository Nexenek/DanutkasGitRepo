Zapytanie 4
SELECT m.nazwa, ROUND(AVG(w.liczba_doroslych)) AS srednia_doroslych FROM miejsca m JOIN wycieczki w ON m.id_miejsca = w.id_miejsca GROUP BY m.nazwa;
